186  subroutine build_d1d_d(DI,II,dgg1,igg1,wm1,b1d,length,nel,nxv,nxp
187  $ ,wi,ti,cb0,cbn)
188 c
189 c Build 1-dimensional multi-element D matrix
190 c I
191 c
192 c
193 c T -1 T
194 c D := D Q B Q D
195 c I GG I I I GG
196 c
197 c
198 c
199 c T -1 T
200 c I := I Q B Q I
201 c I GG I I I GG
202 c
203 c
204 c
205 c Here,
206 c
207 c
208 c
209 c I := K1-block-diagonal matrix of interpolation operators from M2 to M1.
210 c GG
211 c
212 c
213 c D := K1-block-diagonal matrix of derivative operators from M2 to M1.
214 c GG
215 c
216 c T
217 c Q := K1 one-dimensional spectral element direct stiffness summation op.
218 c I
219 c
220 c T
221 c B := K1-block-diagonal (dssumed) 1D mass-matrix (B = Q B Q )
222 c I I I L I
223 c
224 c
225 c Return the matrices, I and D , ready for a gen'd eigenv. solver.
226 c I I
227 c
228  real di(1),ii(1),wi(1),ti(1),dgg1(1),igg1(1),wm1(1),b1d(1)
229  real length(nel)
230  character*1 cb0,cbn
231 c
232 c
233 c Number of active points on mesh 1 depends on the boundary
234 c conditions at z0 and zn:
235 c
236 c If periodic: n1 = nel*(nxv-1) := n1p
237 c
238 c If Dirichlet at both ends, n1 = n1p - 1
239 c (However, we retain both, and use mask...)
240 c
241 c If Neumann (outflow) at both ends, n1 = n1p + 1
242 c
243 c If Dirichlet at one end outflow at other, n1 = n1p
244 c
245 c
246 c *** NOTE that for symmetric boundary conditions ('S' := SYM ) the number
247 c of degrees-of-freedom for the velocity field in the D_GG term is n1p-1,
248 c while the number of degrees-of-freedom in the velocity field for the
249 c I_GG term is n1p+1. This is because the I_GG term operates on the
250 c other *velocity* components (those tangential to the symmetry plane)
251 c for which Neumann boundary conditions are appropriate.
252 c
253  n = nel*nxp
254  n1 = nel*(nxv-1) + 1
255  if (cb0.eq.'P') n1 = n1-1
256 c
257 c
258 c Perform direct summation of 1D mass matrices
259 c
260  i = 1
261  call rzero(b1d,n1+1)
262  do ie=1,nel
263 c build B
264  s = 0.5*length(ie)
265  call add2s2(b1d(i),wm1,s,nxv)
266  i = i + (nxv-1)
267  enddo
268  if (cb0.eq.'P') b1d( 1) = b1d( 1)+b1d(n1+1)
269  call invcol1(b1d,n1)
270 c
271 c Mask velocity mesh at endpoints of 1D operator
272 c
273  if (cb0.eq.'D') b1d( 1) = 0.
274  if (cbn.eq.'D') b1d(n1) = 0.
275 c
276 c
277 c Perform direct summation of 1D I matrices
278 c GG
279  i = 1
280  j = 1
281  call rzero(wi,n*n1)
282  do ie=1,nel
283  s = 0.5*length(ie)
284  call add2s2mat2p(wi,n1,n,i,j,igg1,nxv,s,nxv,nxp)
285  i = i + (nxv-1)
286  j = j + (nxp)
287  enddo
288 c -1 T
289 c Set ti = ( B W )
290 c I I
291  do i=1,n1
292  do j=1,n
293  ij = i + n1*(j-1)
294  ji = j + n*(i-1)
295  ti( ji ) = b1d(i)*wi(ij)
296  enddo
297  enddo
298  call mxm(ti,n,wi,n1,ii,n)
299 c
300 c
301 c Mask velocity mesh at endpoints of 1D operator for SYM+E case
302 c
303  if (cb0.eq.'S') b1d( 1) = 0.
304  if (cbn.eq.'S') b1d(n1) = 0.
305 c
306 c
307 c T
308 c Compute w = Q D (Note: w is (n1 x n) , n := (nxp)*nel )
309 c I I GG I
310 c
311 c
312 c
313 c
314 c Perform direct summation of 1D D matrices
315 c GG
316 c
317  i = 1
318  j = 1
319  call rzero(wi,n*n1)
320  do ie=1,nel
321  s = 1.0
322  call add2s2mat2p(wi,n1,n,i,j,dgg1,nxv,s,nxv,nxp)
323  i = i + (nxv-1)
324  j = j + nxp
325  enddo
326 c
327 c -1 T
328 c Set ti = ( B W )
329 c I I
330  do i=1,n1
331  do j=1,n
332  ij = i + n1*(j-1)
333  ji = j + n*(i-1)
334  ti( ji ) = b1d(i)*wi(ij)
335  enddo
336  enddo
337  call mxm(ti,n,wi,n1,di,n)
338 c
339  return
340  end

342  subroutine gfdm_set_geom(work,melx,mely,melz)
343 c

371  scalez = 1./(nelx*nely)

388  enddo

391  do ez=1,nelz
392  zgtp(ez) = zgtp(ez-1) + lz(ez)
393  enddo
394 
395  return
396  end
