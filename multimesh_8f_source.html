263  subroutine bcopy

265  include 'TOTAL'
266  include 'NEKNEK'

269  n = lx1*ly1*lz1*nelt

550  subroutine exchange_points2(dxf,dyf,dzf)
551  include 'SIZE'
552  include 'TOTAL'
553  include 'NEKUSE'
554  include 'NEKNEK'
555  integer i,j,k,n
556  common /nekmpi/ mid,mp,nekcomm,nekgroup,nekreal
557  integer jsend(nmaxl_nn)
558  common /exchr/ rsend(ldim*nmaxl_nn)
559  integer rcode_all(nmaxl_nn),elid_all(nmaxl_nn)
560  integer proc_all(nmaxl_nn)
561  real dist_all(nmaxl_nn)
562  real rst_all(nmaxl_nn*ldim)
563  integer e,ip,iface,nel,nfaces,ix,iy,iz
564  integer kx1,kx2,ky1,ky2,kz1,kz2,idx,nxyz,nxy
565  integer icalld
566  save icalld
567  data icalld /0/
568 
569 cccc
570 c Look for boundary points with Diriclet b.c. (candidates for
571 c interpolation)
572 
573  ifield = 1
574  if (ifheat) ifield = 2
575 
576  nfaces = 2*ldim
577  nel = nelfld(ifield)
578 
579  nxyz = lx1*ly1*lz1
580  ntot = nxyz*nel
581  nxy = lx1*ly1
582  call izero(imask,ntot)
583 
584 cccc
585 c Setup arrays of x,y,zs to send to findpts and indices of boundary
586 c points in jsend
587  ip = 0
588  if (idsess.eq.0) then
589  dxf = -dxf
590  endif
591  do e=1,nel
592  do iface=1,nfaces
593  if (intflag(iface,e).eq.1) then
594  call facind (kx1,kx2,ky1,ky2,kz1,kz2,lx1,ly1,lz1,iface)
595  do iz=kz1,kz2
596  do iy=ky1,ky2
597  do ix=kx1,kx2
598  call nekasgn (ix,iy,iz,e)
599  ip=ip+1
600  idx = (e-1)*nxyz+(iz-1)*nxy+(iy-1)*lx1+ix
601  jsend(ip) = idx
602  if (if3d) then
603  rsend(ldim*(ip-1)+1)=x-dxf
604  rsend(ldim*(ip-1)+2)=y
605  rsend(ldim*(ip-1)+3)=z
606  else
607  rsend(ldim*(ip-1)+1)=x-dxf
608  rsend(ldim*(ip-1)+2)=y
609  endif
610 
611  if (ip.gt.nmaxl_nn) then
612  write(6,*) nid,
613  & ' ABORT: nbp (current ip) too large',ip,nmaxl_nn
614  call exitt
615  endif
616 
617  imask(idx,1,1,1)=1
618 
619  enddo
620  enddo
621  enddo
622  endif
623  enddo
624  enddo
625  nbp = ip
626 
627  call neknekgsync()
628 
629 c JL's routine to find which points these procs are on
630  call fgslib_findpts(inth_multi2,rcode_all,1,
631  & proc_all,1,
632  & elid_all,1,
633  & rst_all,ldim,
634  & dist_all,1,
635  & rsend(1),ldim,
636  & rsend(2),ldim,
637  & rsend(3),ldim,nbp)
638 
639  call neknekgsync()
640 
641 c Move mesh 1 back to its original position
642  if (idsess.eq.0) then
643  call cadd(xm1,-dxf,lx1*ly1*lz1*nelt)
644  endif
645 
646  ip=0
647  icount=0
648  ierror=0
649 
650 c Make sure rcode_all is fine
651  do 200 i=1,nbp
652 
653  if (rcode_all(i).lt.2) then
654 
655  if (rcode_all(i).eq.1.and.dist_all(i).gt.1e-02) then
656  if (ldim.eq.2) write(6,*)
657  & 'WARNING: point on boundary or outside the mesh xy[z]d^2: '
658  if (ldim.eq.3) write(6,*)
659  & 'WARNING: point on boundary or outside the mesh xy[z]d^2: '
660  goto 200
661  endif
662  ip=ip+1
663  rcode(ip) = rcode_all(i)
664  elid(ip) = elid_all(i)
665  proc(ip) = proc_all(i)
666  do j=1,ldim
667  rst(ldim*(ip-1)+j) = rst_all(ldim*(i-1)+j)
668  enddo
669  ilist(1,ip) = jsend(i)
670 
671  endif ! rcode_all
672 
673  200 continue
674 
675  npoints_nn = ip
676 
677 c zero out valint
678  do i=1,nfld_neknek
679  call rzero(valint(1,1,1,1,i),lx1*ly1*lz1*nelt)
680  enddo
681 
682  call iglmax(ierror,1)
683  if (ierror.eq.1) call exitt
684 
685  call neknekgsync()
686 
687  return
688  end
