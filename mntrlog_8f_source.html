211  endif
212  endif
213  endif
214 
215 ! check convergence flag
216  if (mntr_ifconv.and.(nsteps-istep).gt.lstdl) then
217  call mntr_log(mntr_id,lp_inf,
218  $ 'Simulation converged; adjust NSTEPS')
219  nsteps = istep+lstdl
220  endif
221 
222

414  ipos = -il

420  if (ipos.eq.0) then

422  if (mntr_mod_id(il).eq.-1) then
423  ipos = il
424  exit
425  endif
426  enddo
427  endif
428  endif
429 
430 ! broadcast mid
431  call bcast(ipos,isize)
432 
433 ! error; no free space found
434  if (ipos.eq.0) then
435  mid = ipos
436  call mntr_abort(mntr_id,
437  $ 'module ['//trim(lname)//'] cannot be registered')
438 ! module already registered
439  elseif (ipos.lt.0) then
440  mid = abs(ipos)
441  call mntr_abort(mntr_id,
442  $ 'Module ['//trim(lname)//'] is already registered')
443 ! new module
444  else
445  mid = ipos
446 ! check if parent module is registered
447  if (pmid.gt.0) then
448  if (mntr_mod_id(pmid).ge.0) then
449  mntr_mod_id(ipos) = pmid
450  else
451  mntr_mod_id(ipos) = 0
452  call mntr_log(mntr_id,lp_inf,
453  $ "Module's ["//trim(lname)//"] parent not registered.")
454  endif
455  else
456  mntr_mod_id(ipos) = 0
457  endif
458  mntr_mod_name(ipos)=lname
459  mntr_mod_dscr(ipos)=ldscr
460  mntr_mod_num = mntr_mod_num + 1
461  if (mntr_mod_mpos.lt.ipos) mntr_mod_mpos = ipos
462  call mntr_log(mntr_id,lp_inf,
463  $ 'Registered module ['//trim(lname)//']: '//trim(ldscr))
464  endif
465 
466  return
467  end subroutine
