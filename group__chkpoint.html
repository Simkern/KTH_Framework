Module interface usage:
subroutine userchk
include 'TSTEP'
if (istep.eq.0) then
! start framework
call frame_start
! register modules
call chkpt_register
! get runtime parameters
call frame_rparam
! initialise modules
call chkpt_init
endif
! monitor simulation
call frame_monitor
! save/load files for full-restart
call chkpt_main
! finalise framework
if (istep.ge.nsteps.or.lastep.eq.1) then
call frame_end
endif
return
end
