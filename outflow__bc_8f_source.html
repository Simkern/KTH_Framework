94 !******************************************************************************
95 !
96 ! Purpose: set outflow boundary condition for one cell.
97 !
98 ! Description: the subsonic boundary condition is based on non-reflecting,
99 ! characteristics method of Whitfield and Janus: Three-Dimensional
100 ! Unsteady Euler Equations Solution Using Flux Vector Splitting.
101 ! AIAA Paper 84-1552, 1984. The supersonic boundary condition
102 ! consists of simple extrapolation.
103 !
104 ! Input: bcOpt = boundary treatment: subsonic, supersonic, or mixed
105 ! pout = given static outlet pressure
106 ! sx/y/zn = components of ortho-normalized face vector (outward facing)
107 ! cpgas = specific heat at constant pressure (boundary cell)
108 ! mol = molecular mass at boundary cell
109 ! rho = density at boundary cell
110 ! rhou/v/w = density * velocity components at boundary cell
111 ! rhoe = density * total energy at boundary cell
112 ! press = static pressure at boundary cell
113 !
114 ! Output: rhob = density at boundary
115 ! rhou/v/wb = density * velocity components at boundary
116 ! rhoeb = density * total energy at boundary
117 !
118 ! Notes: this condition is valid only for thermally and calorically
119 ! perfect gas (supersonic outflow valid for all gases).
120 !
121 !******************************************************************************
122 !
123 ! $Id: BcondOutflowPerf.F90,v 1.1.1.1 2014/05/05 21:47:47 tmish Exp $
124 !
125 ! Copyright: (c) 2002 by the University of Illinois
126 !
127 !******************************************************************************
128 
129  SUBROUTINE bcondoutflowperf( bcOpt,pout,sxn,syn,szn,cpgas,mol,
130  > rho,rhou,rhov,rhow,rhoe,press,
131  > rhob,rhoub,rhovb,rhowb,rhoeb )
132 
133  IMPLICIT NONE
134  integer bcopt_subsonic,bcopt_mixed
135  parameter(bcopt_subsonic=1)
136  parameter(bcopt_mixed=0)
137  real MixtPerf_C_DGP,MixtPerf_Eo_DGPUVW,MixtPerf_G_CpR,MixtPerf_R_M
138  > , mixtperf_p_deogvm2
139  external mixtperf_c_dgp,mixtperf_eo_dgpuvw,mixtperf_g_cpr,
140  > mixtperf_r_m,mixtperf_p_deogvm2
141 
142 ! ... parameters
143  INTEGER bcOpt
144 
145  REAL pout
146  REAL rho, rhou, rhov, rhow, rhoe, press
147  REAL sxn, syn, szn, cpgas, mol
148  REAL rhob, rhoub, rhovb, rhowb, rhoeb
149 
150 ! ... local variables
151  REAL csound, rgas, gamma, gam1, u, v, w, mach, rrhoc, deltp,
152  > ub, vb, wb, vnd
153 
154 !******************************************************************************
155 ! gas properties; velocity components; Mach number
156 
157  rgas = mixtperf_r_m( mol )
158  gamma = mixtperf_g_cpr( cpgas,rgas )
159  gam1 = gamma - 1.0
160 
161  u = rhou/rho
162  v = rhov/rho
163  w = rhow/rho
164  csound = mixtperf_c_dgp( rho,gamma,press )
165  mach = sqrt(u*u+v*v+w*w)/csound
166 
167 ! subsonic flow ---------------------------------------------------------------
168 
169  IF (mach .lt. 1.0 .AND.
170  >(bcopt .eq. bcopt_subsonic .OR. bcopt .eq. bcopt_mixed)) THEN
171  rrhoc = 1.0/(rho*csound)
172  deltp = press - pout
173  rhob = rho - deltp/(csound*csound)
174  ub = u + sxn*deltp*rrhoc
175  vb = v + syn*deltp*rrhoc
176  wb = w + szn*deltp*rrhoc
177 
178 ! - special treatment to prevent "deltp" from changing the sign
179 ! of velocity components. This may happen for very small u, v, w.
180 
181  vnd = ub*sxn + vb*syn + wb*szn
182  IF ( vnd .lt. 0.0 ) THEN ! inflow at outflow boundary
183  ub = sign(1.0,u)*max(abs(ub),abs(u))
184  vb = sign(1.0,v)*max(abs(vb),abs(v))
185  wb = sign(1.0,w)*max(abs(wb),abs(w))
186  END IF ! vnd
187 
188  rhoub = rhob*ub
189  rhovb = rhob*vb
190  rhowb = rhob*wb
191  rhoeb = rhob*mixtperf_eo_dgpuvw( rhob,gamma,pout,ub,vb,wb )
192 
193 ! supersonic flow -------------------------------------------------------------
194 
195  ELSE
196  rhob = rho
197  rhoub = rhou
198  rhovb = rhov
199  rhowb = rhow
200  rhoeb = rhoe
201  END IF ! mach
202 
203  end
