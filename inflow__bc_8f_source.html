128  do ix=i0,i1
129  call nekasgn(ix,iy,iz,e)

131 c ! ux,uy,uz
132  l=l+1
133  wbc(l,f,e,irho) = rho ! Dirichlet, userbc
134  wbc(l,f,e,iux) = ux ! Dirichlet, userbc
135  wbc(l,f,e,iuy) = uy ! Dirichlet, userbc
136  wbc(l,f,e,iuz) = uz ! Dirichlet, userbc
137  wbc(l,f,e,iph) = phi ! Dirichlet, userbc
138  rhob = rho*phi
139  rhoub = rho*ux*phi
140  rhovb = rho*uy*phi
141  rhowb = rho*uz*phi
142  wbc(l,f,e,iu1) = rhob
143  wbc(l,f,e,iu2) = rhoub
144  wbc(l,f,e,iu3) = rhovb
145  wbc(l,f,e,iu4) = rhowb
146 
147  if (if3d) then ! shouldn't this be normal Mach number?
148  mach = sqrt(ux**2+uy**2+uz**2)/asnd
149  snz = unz(l,1,f,e)
150  else
151  mach = sqrt(ux**2+uy**2)/asnd
152  snz=0.0
153  endif
154 
155  snx = unx(l,1,f,e)
156  sny = uny(l,1,f,e)
157 
158  if (mach.lt.1.0) then
159 
160  pres = facew(l,f,e,ipr) ! extrapolated, overwritten
161  temp = pres/rho/(cp-cv) ! definitely too perfect!
162  wbc(l,f,e,ipr) = pres
163  wbc(l,f,e,isnd) = sqrt(cp/cv*pres/rho) ! too perfect?
164  wbc(l,f,e,ithm) = temp ! definitely too perfect!
165  wbc(l,f,e,icpf) = rho*cp ! NEED EOS WITH TEMP Dirichlet, userbc
166  wbc(l,f,e,icvf) = rho*cv ! NEED EOS WITH TEMP Dirichlet, userbc
167 
168  else ! supersonic inflow
169 
170  wbc(l,f,e,ipr) = pres
171  wbc(l,f,e,isnd) = asnd
172  wbc(l,f,e,ithm) = temp
173  wbc(l,f,e,icpf) = rho*cp
174  wbc(l,f,e,icvf) = rho*cv
175 
176  endif
177 
178 ! find a smarter way of doing this. fold it into usr file if you must
179  wbc(l,f,e,iu5) = phi*rho*cv*temp+0.5/rhob*(rhoub**2+rhovb**2+
180  > rhowb**2)
181 
182  enddo
183  enddo
184  enddo
185 
186  return
187  end

242 !
243 ! Purpose: set inflow boundary condition for one cell.
244 !
245 ! Description: The subsonic boundary condition is based on the extrapolation of
246 ! the Riemann invariant from the interior field (Holmes, D.G.: Inviscid 2D
247 ! Solutions on Unstructured, Adaptive Grids. VKI Lecture Series 1989-06, 1989).
248 ! The supersonic inflow boundary condition computes the conservative variables
249 ! from given velocity components, density and pressure.
250 !
251 ! Input: bcOptType = boundary treatment: subsonic, supersonic, or mixed
252 ! bcOptFixed = whether _computed_ inflow angle should be fixed or not
253 ! ptot = given total pressure
254 ! ttot = given total temperature
255 ! sx/y/zn = components of normalized face vector (outward facing)
256 ! cpgas = specific heat at constant pressure (boundary cell)
257 ! mm = molecular mass at boundary cell
258 ! rl = given density
259 ! ru/v/wl = given velocity components
260 !
261 ! Output: rr = density at boundary | velocity compts inout from userbc
262 ! ru/v/wr = density * velocity components at boundary
263 ! rer = density * total internal energy at boundary
264 ! pr = pressure at boundary
265 !
266 ! Notes:
267 ! 1. This condition is valid only for thermally and calorically perfect
268 ! gas.
269 ! 2. Important to avoid division by MakeNonZero(sl) when computing eta
270 ! because that computation can become undefined for reservoir inflow
271 ! conditions, i.e., for a vanishing velocity vector.
272 !
273 !******************************************************************************
274 !

281  SUBROUTINE bcondinflowperf(bcOptType,bcOptFixed,ptot,ttot,ur,vr,wr
282  > ,mach,sxn,syn,szn,cpgas,mm,rl,rul
283  > ,rvl,rwl,rr,rur,rvr,rwr,rer,pr,ar,tr)
284  IMPLICIT NONE
285 
286  integer BCOPT_SUBSONIC, BCOPT_MIXED, BCOPT_FIXED_NO
287 ! faked flags
288  parameter(bcopt_subsonic = 1)
289  parameter(bcopt_mixed = 2)
290  parameter(bcopt_fixed_no = 2)
291  real MixtPerf_C_Co2GUVW, MixtPerf_C_DGP, MixtPerf_C_GRT,
292  > mixtperf_co2_cguvw, mixtperf_c2_grt,mixtperf_d_prt,
293  > mixtperf_eo_dgpuvw, mixtperf_eo_dgpvm, mixtperf_g_cpr,
294  > mixtperf_p_gmapo, mixtperf_p_gpotto, mixtperf_po_gptto,
295  > mixtperf_po_cgpuvw, mixtperf_r_m, mixtperf_t_cgr,
296  > mixtperf_t_gmato, mixtperf_vm_c2co2g
297  external mixtperf_c_co2guvw, mixtperf_c_dgp, mixtperf_c_grt,
298  > mixtperf_co2_cguvw, mixtperf_c2_grt,mixtperf_d_prt,
299  > mixtperf_eo_dgpuvw, mixtperf_eo_dgpvm, mixtperf_g_cpr,
300  > mixtperf_p_gmapo, mixtperf_p_gpotto, mixtperf_po_gptto,
301  > mixtperf_po_cgpuvw, mixtperf_r_m, mixtperf_t_cgr,
302  > mixtperf_t_gmato, mixtperf_vm_c2co2g
303 
304 
305 ! ... parameters
306  INTEGER bcOptFixed,bcOptType!, INTENT(IN) ::
307 ! . . . . . . from userbc
308  REAL cpgas, mach, mm, sxn, syn, szn, ur, vr, wr,!, INTENT(IN) ::
309  > ptot, rl, rul, rvl, rwl, ttot
310 
311  REAL rer, rr, rur, rvr, rwr, pr!, INTENT(OUT) ::
312 
313  REAL al, ar, a02, cp, disc, eta, g, gm1, igm1, ql, rgas, Rm,
314  > sl, sr, tr, ul, vl, wl
315 
316 !******************************************************************************
317 ! gas properties
318 
319  rgas = mixtperf_r_m(mm)
320  g = mixtperf_g_cpr(cpgas,rgas)
321 
322 ! subsonic or mixed -----------------------------------------------------------
323 
324  IF ( bcopttype.eq.bcopt_subsonic.OR.bcopttype.eq.bcopt_mixed) THEN
325  gm1 = g - 1.0
326  igm1 = 1.0/gm1
327 
328  ul = rul/rl
329  vl = rvl/rl
330  wl = rwl/rl
331 
332  a02 = mixtperf_c2_grt(g,rgas,ttot)
333 
334  al = mixtperf_c_co2guvw(a02,g,ul,vl,wl) ! make al consistent with a02
335  ql = ul*sxn + vl*syn + wl*szn
336 
337 ! - subsonic
338 
339  IF ( abs(ql) .lt. al ) THEN
340  sl = sqrt(ul*ul + vl*vl + wl*wl)
341 
342  IF ( bcoptfixed .eq. bcopt_fixed_no ) THEN
343  IF ( sl .gt. 1.0e-6 ) THEN ! Avoid ill-defined angle computation
344  eta = ql/sl
345  ELSE
346  eta = -1.0
347  END IF ! sl
348  ELSE
349  eta = -1.0
350  END IF ! bcOptFixed
351 
352  rm = al - 0.5*gm1*ql
353  disc = 0.5*gm1*eta**2*
354  > (a02/(rm*rm)*(1.0 + 0.5*gm1*eta**2) - 1.0)
355 
356  IF ( disc .lt. 0.0 ) THEN ! discriminant cannot be negative
357  ar = sqrt(a02)
358  tr = ttot
359  pr = ptot
360  sr = 0.0
361  ELSE
362  ar = rm/(1.0 + 0.5*gm1*eta*eta)*(1.0+sqrt(disc))
363  tr = mixtperf_t_cgr(ar,g,rgas)
364  pr = mixtperf_p_gpotto(g,ptot,tr,ttot)
365  sr = mixtperf_vm_c2co2g(ar*ar,a02,g)
366  END IF ! disc
367 
368  rr = mixtperf_d_prt( pr,rgas,tr )
369 
370  rer = rr*mixtperf_eo_dgpvm(rr,g,pr,sr)
371  rur = rr*ur
372  rvr = rr*vr
373  rwr = rr*wr
374 
375 ! - supersonic
376 
377  ELSE
378  IF ( bcopttype .eq. bcopt_mixed ) THEN
379  pr = mixtperf_p_gmapo(g,mach,ptot)
380  tr = mixtperf_t_gmato(g,mach,ttot)
381  rr = mixtperf_d_prt(pr,rgas,tr)
382  ar = mixtperf_c_grt(g,rgas,tr)
383 
384  rer = rr*mixtperf_eo_dgpuvw(rr,g,pr,ur,vr,wr)
385  rur = rr*ur
386  rvr = rr*vr
387  rwr = rr*wr
388  END IF ! bcOptType
389  END IF ! ql < al
390 
391 ! supersonic ------------------------------------------------------------------
392 
393  ELSE ! bcOptType .eq. BCOPT_SUPERSONIC
394  pr = mixtperf_p_gmapo(g,mach,ptot)
395  tr = mixtperf_t_gmato(g,mach,ttot)
396  rr = mixtperf_d_prt(pr,rgas,tr)
397  ar = mixtperf_c_grt(g,rgas,tr)
398 
399  rer = rr*mixtperf_eo_dgpuvw(rr,g,pr,ur,vr,wr)
400  rur = rr*ur
401  rvr = rr*vr
402  rwr = rr*wr
403  END IF ! bcOptType
404 
405  return
406  end
